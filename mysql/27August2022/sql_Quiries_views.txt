SELECT customerNumber, customerName, city,country, employeeNumber, lastName, email, jobTitle FROM customers, employees WHERE employees.employeeNumber = customers.salesRepEmployeeNumber;


## field ak hole avabe korte hbe from ar por customer k rename kora hoice ##

SELECT 
c.customerNumber, c.customerName, c.city, c.country, 
e.employeeNumber, e.lastName, e.email, e.jobTitle 
FROM customers c, employees e WHERE e.employeeNumber = c.salesRepEmployeeNumber;

## Create view relational data ##

CREATE VIEW customerslist_with_salesRep_view AS
SELECT 
c.customerNumber, c.customerName, c.city, c.country, 
e.employeeNumber, e.lastName, e.email, e.jobTitle 
FROM customers c, employees e WHERE e.employeeNumber = c.salesRepEmployeeNumber;
